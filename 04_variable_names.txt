dirName, destName - source_dir, dest_dir
// имена каталогов с исходными данными и результатами обработки, (destName - в конексте с dirName легко принять за имя файла)

iD20 - inv_lead_coef_matrix
// обратная матрица к матрице коэффициентов при старшей производной в системе дифференциальных уравнений

l - orbital_momentum
// орбитальный момент импульса (в квантовой механике - безразмерная величина и всегда целое число)

result - poly_power
// результат возведения многочлена в степень

result - solutions
// список кортежей с параметрами для разных решений дифференциального уравнения

waves - radial_funcs
// массив радиальных функций, задающих волновую функцию (совсем не волны, и даже не волновые функции)

zxList - ccp_charges
// массив (а вовсе не список) зарядовых чисел, характеризующих потенциал центральной ячейки (central cell potential - ccp)

lxList - ccp_ranges_nm
// массив значений области действия CCP, измеряемых в нанометрах

index - level_index
// номер уровня

energies - energies_meV
// сетка энергий, измеряемых в миллиэлектронвольтах

r - r_mesh_nm
// сетка длин радиус-вектора, измеряемых в нанометрах
